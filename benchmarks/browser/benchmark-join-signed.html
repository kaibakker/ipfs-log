ipfs-log - benchmark join()

Description

Add an entry to two logs and join them after each update. Measure throughput in operations per second.

    const keystore = Keystore.create('./browser-benchmark-join-test-keys')
    const key = keystore.createKey('browser-benchmark-join-signed')
    const provider = new IdentityProvider(
      data => keystore.sign(key, data),
      async (sig, entryKey, data) =>  {
        const pubKey = await keystore.importPublicKey(entryKey)
        return keystore.verify(sig, pubKey, data)
      }
    )
    const acl = new ACL((pubKey, entry) => Promise.resolve(pubKey === key.getPublic('hex')))
    const identity = new Identity(
      key.getPublic('hex'),
      key.getPublic('hex'),
      provider
    )

    log1 = new Log(ipfs, 'browser-benchmark-join-signed-A', null, null, null, acl, identity)
    log2 = new Log(ipfs, 'browser-benchmark-join-signed-B', null, null, null, acl, identity)

    const add1 = await log1.append("a" + loopCount)
    const add2 = await log2.append("b" + loopCount)

    Promise.all([add1, add2])
      .then((res) => {
        log1.join(log2)
        log2.join(log1)
      })
